```python
def fibonacci(n: int) -> int:
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç n-–µ —á–∏—Å–ª–æ –§–∏–±–æ–Ω–∞—á—á–∏ –Ω–∞–∏–≤–Ω—ã–º —Ä–µ–∫—É—Ä—Å–∏–≤–Ω—ã–º —Å–ø–æ—Å–æ–±–æ–º.
```   

```python
def fibonacci_memo(n: int, memo: Optional[Dict[int, int]] = None) -> int:
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç n-–µ —á–∏—Å–ª–æ –§–∏–±–æ–Ω–∞—á—á–∏ —Å –º–µ–º–æ–∏–∑–∞—Ü–∏–µ–π.

        memo: –°–ª–æ–≤–∞—Ä—å –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –≤—ã—á–∏—Å–ª–µ–Ω–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π

    Returns:
        n-–µ —á–∏—Å–ª–æ –§–∏–±–æ–Ω–∞—á—á–∏
    """
    if memo is None:
        memo = {}

    if n in memo:
        return memo[n]

    if n == 0:
        return 0
    if n == 1:
        return 1

    memo[n] = fibonacci_memo(n - 1, memo) + fibonacci_memo(n - 2, memo)
    return memo[n]
```

```python
def fast_power(a: float, n: int) -> float:
    """
    –ë—ã—Å—Ç—Ä–æ–µ –≤–æ–∑–≤–µ–¥–µ–Ω–∏–µ —á–∏—Å–ª–∞ a –≤ —Å—Ç–µ–ø–µ–Ω—å n —á–µ—Ä–µ–∑ —Å—Ç–µ–ø–µ–Ω—å –¥–≤–æ–π–∫–∏.

    Args:
        a: –û—Å–Ω–æ–≤–∞–Ω–∏–µ
        n: –ü–æ–∫–∞–∑–∞—Ç–µ–ª—å —Å—Ç–µ–ø–µ–Ω–∏ (—Ü–µ–ª–æ–µ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–æ–µ —á–∏—Å–ª–æ)

    Returns:
        a –≤ —Å—Ç–µ–ø–µ–Ω–∏ n

    Raises:
        ValueError: –ï—Å–ª–∏ n –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–æ–µ
    """
    if n < 0:
        raise ValueError('–ü–æ–∫–∞–∑–∞—Ç–µ–ª—å —Å—Ç–µ–ø–µ–Ω–∏ –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º')
    if n == 0:
        return 1
    if n == 1:
        return a

    half_power = fast_power(a, n // 2)
    if n % 2 == 0:
        return half_power * half_power
    else:
        return a * half_power * half_power
```
